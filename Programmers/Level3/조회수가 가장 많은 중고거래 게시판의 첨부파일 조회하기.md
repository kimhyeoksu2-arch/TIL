## CONCAT 함수
여러 문자열이나 컬럼의 값을 하나로 합칠 때 사용합니다.

## WHERE 절에 서브쿼리를 사용하여 푼 방식

SELECT 
    CONCAT('/home/grep/src/', BOARD_ID, '/', FILE_ID, FILE_NAME, FILE_EXT) AS FILE_PATH
FROM 
    USED_GOODS_FILE
WHERE 
    BOARD_ID = (
        SELECT BOARD_ID 
        FROM USED_GOODS_BOARD 
        ORDER BY VIEWS DESC 
        LIMIT 1
    )
ORDER BY 
    FILE_ID DESC;
